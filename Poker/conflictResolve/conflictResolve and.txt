conflictResolve: hand1 and: hand2	| type |		type := hand1 getType.		type == 10 ifTrue: [ ^self checkRoyalFlush: hand1 against: hand2. ].	type == 9 ifTrue: [ ^self checkStraightFlush: hand1 against: hand2. ].	type == 8 ifTrue: [ ^self checkFourOfAKind: hand1 against: hand2. ].	type == 7 ifTrue: [ ^self checkFullHouse: hand1 against: hand2. ].	type == 6 ifTrue: [ ^self checkFlush: hand1 against: hand2. ].	type == 5 ifTrue: [ ^self checkStraight: hand1 against: hand2. ].	type == 4 ifTrue: [ ^self checkThreeOfAKind: hand1 against: hand2. ].	type == 3 ifTrue: [ ^self checkTwoPair: hand1 against: hand2. ].	type == 2 ifTrue: [ ^self checkOnePair: hand1 against: hand2. ].	^self checkHighCard: hand1 against: hand2.
