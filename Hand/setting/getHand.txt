getHand	| output modifiedHand |		output := OrderedCollection new.	modifiedHand := OrderedCollection new.			1 to: 5 do: [ :i | 		((hand at: i) \\ 13) == 12 ifTrue: [ 			modifiedHand add: ((hand at: i) - 12)		] ifFalse: [ 			modifiedHand add: ((hand at: i) + 1)		]	].		1 to: 5 do: [ :i |		((modifiedHand at: i) // 13) == 0 ifTrue: [ 			output add: ( '{1}C' format: { ((modifiedHand at: i) \\ 13)+1 } )		].		((modifiedHand at: i) // 13) == 1 ifTrue: [ 			output add: ( '{1}D' format: { ((modifiedHand at: i) \\ 13)+1 } )		].		((modifiedHand at: i) // 13) == 2 ifTrue: [ 			output add: ( '{1}H' format: { ((modifiedHand at: i) \\ 13)+1 } )		].		((modifiedHand at: i) // 13) == 3 ifTrue: [ 			output add: ( '{1}S' format: { ((modifiedHand at: i) \\ 13)+1 } )		].	].		^output.		"^hand."
